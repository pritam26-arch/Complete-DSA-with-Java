ALGORITHM: Sum of Two Numbers

START
    // Step 1: Initialize variables
    DECLARE num1, num2, sum

    // Step 2: Take input from the user
    INPUT num1
    INPUT num2

    // Step 3: Perform the logic (Processing)
    sum = num1 + num2

    // Step 4: Output the result
    PRINT "The sum is: " + sum
END
